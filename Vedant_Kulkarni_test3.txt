#include<iostream>
#include<vector>
using namespace std;
class Employee
{
    public:
        virtual double calculateSalary() = 0;
};
class PermanentEmployee: public Employee
{
    private:
        double base, bonus;
    public:
        PermanentEmployee(double base, double bonus)
        {
            this->base = base;
            this->bonus = bonus;
        }
        double calculateSalary()
        {
            return base + bonus;
        }
};
class ContractualEmployee: public Employee
{
    private:
        double hourlyrate, hoursworked;
    public:
        ContractualEmployee(double hourlyrate, double hoursworked)
        {
            this->hourlyrate = hourlyrate;
            this->hoursworked = hoursworked;
        }
        double calculateSalary()
        {
            return hourlyrate*hoursworked;
        }
};
int main()
{
    vector<Employee*> e1;
    Employee* emp1 = new PermanentEmployee(50000, 2000);
    e1.push_back(emp1);
    Employee* emp2 = new ContractualEmployee(700, 300);
    e1.push_back(emp2);
    for(int i =0; i<2; i++)
    {
        cout<<e1[i]->calculateSalary()<<endl;
    }
    return 0;
}
